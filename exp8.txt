sudo apt-get install docker
sudo docker pull ubuntu
sudo docker images
sudo docker run -it -d ubuntu ----> runs ubuntu with this command
sudo docker ps ---> lists running container on system
sudo docker ps -a ---->This command list all containers running or exited from the system.
sudo docker exec -it (container_id of image u want to execute) bash
do ls in it
run some basic command for that image if ubuntu then cat,touch,mkdir etc
write exit
